# Laravel Production Cron Jobs
# Add these to your crontab: crontab -e

# Laravel scheduler (runs every minute)
* * * * * cd /var/www/html/laravelAdminLte3 && /usr/bin/php artisan schedule:run >> /dev/null 2>&1

# Clear logs older than 7 days (runs daily at 3 AM)
0 3 * * * find /var/www/html/laravelAdminLte3/storage/logs -name "*.log" -type f -mtime +7 -delete

# Backup database (runs daily at 2 AM)
0 2 * * * cd /var/www/html/laravelAdminLte3 && /usr/bin/php artisan backup:run >> /dev/null 2>&1

# Optimize application (runs weekly on Sunday at 1 AM)
0 1 * * 0 cd /var/www/html/laravelAdminLte3 && /usr/bin/php artisan optimize

# Clear expired sessions (runs daily at 4 AM)
0 4 * * * cd /var/www/html/laravelAdminLte3 && /usr/bin/php artisan session:gc

# Queue worker health check (runs every 5 minutes)
*/5 * * * * cd /var/www/html/laravelAdminLte3 && /usr/bin/php artisan queue:monitor --max=5 >> /dev/null 2>&1
